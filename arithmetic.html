统计一个字符串出现最多的字母

var str = "abaasdffggghhjjkkgfddsssss3444343";

function getMaxTimeStr(str) {
    var strObj = {};

    if(str.length == 1) {
        return {str: 1};
    }

    for(var i = 0; i < str.length; i++) {
        if(!strObj[str[i]]) {
            strObj[str[i]] = 1;
        } else {
            strObj[str[i]]++;
        }
    }

    var max = 1;
    var maxVal = "";

    for(var k in strObj) {
        if(strObj[k] > max) {
            max = strObj[k];
            maxVal = k;
        }
    }

    return {maxVal: max};
}

数组去重

let arr = [1, 13, 24, 1, 11, 11, 1, 14, 2];

const unique = function(arr) {
    let flag,
        data = [],
        i = 0,
        l = arr.length;

    for(; i < l; i++) {
        flag = true;
        for(let j = 0; j < data.length; j++) {
            if(arr[i] === data[j]) {
                flag = false;
                break;
            }
        }
        if(flag) {
            data[data.length] = arr[i];
        }
    }
    return data;
}

斐波那契数列

const fib = function (num) {
    return num <= 2 ? 1 : fib(num - 1) + fib(num - 2);
}

fib(50);

// 斐波那契数列的递归优化
const fib = function() {
    let cache = [];

    return function (n) {
        if(cache[n] !== undefined) {
            return cache[n];
        }

        if(n === 1 || n === 2) {
            cache[n] = 1;
            return 1;
        }

        let tempNum = arguments.callee(n - 1) + arguments.callee(n - 2);
        cache[n]  = tempNum;

        return tempNum;
    };
};

const fn = fib();
let num = fn(1476);

console.log(num);

冒泡排序

function sortBubble(array) {
var len = array.length, i, j, tmp;
for (i = len - 1; i >= 1; i--) {
        for (j = 0; j < len - i - 1; j++) {
            if(array[j] < array[j + 1]) {
                tmp = array[j];
                array[j] = array[j + 1];
                array[j + 1] = tmp;
            }
        }
    }
    return array;
}

function bubbleSort(arr) {  
    for(let i = 0,l=arr.length;i<l-1;i++) {        
        for(let j = i+1;j<l;j++) { 
            if(arr[i]>arr[j]) {
                    let tem = arr[i];
                    arr[i] = arr[j];
                    arr[j] = tem;
                }
        }
    }    
    return arr;
}

快速排序

function sortQuick(array) {
    var low = 0, high = array.length - 1;
    var sort = function (low, high) {
        if (low == high) {
            return;
        }
        var key = array[low];
        var tmplow = low;
        var tmphigh = high;
        while (low < high) {
            while (low < high && key <= array[high]) {
                --high;
            }
            array[low] = array[high];
            while (low < high && array[low] <= key) {
                ++low;
            }
            array[high] = array[low];
            if (low == tmplow) {
                sort(++low, tmphigh);
                return;
            }
        }
        ;
        array[low] = key;
        sort(tmplow, low - 1);
        sort(high + 1, tmphigh);
    };
    sort(low, high);
    sort();
    return array;
}

插入排序

function sortInsert(array) {
    var i = 1, j, len = array.length, key;
    for (; i < len; i++) {
        j = i;
        key = array[j];
        while (--j > -1) {
            if (array[j] > key) {
                array[j + 1] = array[j];
            }
            else {
                break;
            }
        }
        array[j + 1] = key;
    }
    return array;
}

计算一个时间是当年的第几天

function countDates(year, mon, day) {
    let calc = day;
    let month = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];

    (year % 4 === 0 && year % 100 !== 0) || year % 400 === 0 ? month[1] = 29 : month[1] = 28;

    for(let i = 0; i < mon - 1; i++) {
        calc += month[i];
    }

    return calc;
}

angular1 封装jsonp跨域

angular.module('demo.jsonp', [])
.service('jsonpSrv', ['$window', function ($window) {
    this.jsonp = function (url, params, callback) {
        var arr = [url];
        arr.push('?');
        for(var k in params) {
            arr.push(k + '=' + params[k] + '&');
        }
        var cName = 'myCallback' + Date().now();
        arr.push('callback=' + cName);
        var doc = $window.document;
        var script = doc.createElement('script');
        script.src = arr.join('');
        doc.body.appendChild(script);
        $window[cName] = function (data) {
            callback(data);
            doc.body.removeChild(script);
            delete $window[cName];
        }
    }
}]);
